from langchain.chat_models import ChatOpenAI  # OpenAIВЮў ChatGPT вфевЇИВЮё ВѓгВџЕьЋўЖИ░ ВюёьЋю ьЂ┤въўВіц
from langchain_core.prompts import ChatPromptTemplate  # В▒ёьїЁ ьћёвАгьћёьіИ ьЁюьћївд┐ ВЃЮВё▒ВЮё ВюёьЋю ьЂ┤въўВіц

from langchain_community.document_loaders import PyPDFLoader  # PDF ьїїВЮ╝ВЮё вАювЊюьЋўЖИ░ ВюёьЋю ьЂ┤въўВіц
from langchain.text_splitter import RecursiveCharacterTextSplitter  # ьЁЇВіцьіИвЦ╝ В▓ГьЂгвАю вХёьЋаьЋўЖИ░ ВюёьЋю ьЂ┤въўВіц

# .env ьїїВЮ╝ВЌљВёю ьЎўЖ▓й в│ђВѕў вАювЊю

# ьЎўЖ▓й в│ђВѕўВЌљВёю OpenAI API ьѓц Ж░ђВаИВўцЖИ░

    raise ValueError("OPENAI_API_KEY ьЎўЖ▓й в│ђВѕўвЦ╝ В░ЙВЮё Вѕў ВЌєВіхвІѕвІц. .env ьїїВЮ╝ВЮё ьЎЋВЮИьЋ┤ВБ╝ВёИВџћ.")

# В▓Ф в▓ѕВДИ PDF ьїїВЮ╝ВЮў ВаёВ▓┤ Ж▓йвАю ВЃЮВё▒

pages = loader.load_and_split()

# ьЁЇВіцьіИ вХёьЋаЖИ░ ВёцВаЋ - ьЂ░ В▓ГьЂг ВѓгВЮ┤ВдѕвАю ВёцВаЋьЋўВЌг В╗еьЁЇВіцьіИ ВюаВДђ

    chunk_size=4000,  # Ж░Ђ В▓ГьЂгВЮў Вхювїђ вгИВъљ Вѕў
    chunk_overlap=400,  # В▓ГьЂг Ж░ё ВцЉв│хвљўвіћ вгИВъљ Вѕў
    length_function=len
)

# ьјўВЮ┤ВДђвЊцВЮё В▓ГьЂгвАю вХёьЋа

# ьЁЇВіцьіИвЦ╝ Ваёв░ўвХђВЎђ ьЏёв░ўвХђ вЉљ Ж░юВЮў ьїїьіИвАю в│ЉьЋЕ
if len(texts) > 2:
    # Ваёв░ўвХђ ьЁЇВіцьіИ в│ЉьЋЕ
    first_half = " ".join([t.page_content for t in texts[:len(texts)//2]])
    # ьЏёв░ўвХђ ьЁЇВіцьіИ в│ЉьЋЕ
    second_half = " ".join([t.page_content for t in texts[len(texts)//2:]])
    texts = [
        {"content": first_half, "part": "Ваёв░ўвХђ"},
        {"content": second_half, "part": "ьЏёв░ўвХђ"}
    ]

llm = ChatOpenAI(
    openai_api_key=openai_api_key,
    model="gpt-3.5-turbo",  # GPT-3.5 вфевЇИ ВѓгВџЕ
    temperature=0.1,  # вѓ«ВЮђ temperatureвАю ВЮ╝Ж┤ђвљю ВХюваЦ ВЃЮВё▒
    max_tokens=1000  # ВХЕвХёьЋю ЖИИВЮ┤ВЮў ВџћВЋйВЮё ВюёьЋю ьєаьЂ░ Вѕў ВёцВаЋ
)

# ВџћВЋйВЮё ВюёьЋю ьћёвАгьћёьіИ ьЁюьћївд┐ ВёцВаЋ
prompt = ChatPromptTemplate.from_messages([
    ("system", "вІ╣ВІаВЮђ ВаёвгИ ьЋЎВѕа ВЌ░ЖхгВЏљВъЁвІѕвІц. вІцВЮї ьЋЎВѕа вЁ╝вгИ ьїїьіИвЦ╝ ВБ╝Вџћ ьЈгВЮИьіИВЎђ ьЋхВІг в░юЖ▓гВѓгьЋГВЮё ВцЉВІгВю╝вАю ьЋюЖИђвАю ьЈгЖ┤ёВаЂВЮ┤Ж│а ВЃЂВёИьЋўЖ▓ї ВџћВЋйьЋ┤ВБ╝ВёИВџћ."),
    ("user", "вІцВЮї ьЁЇВіцьіИвЦ╝ ьЋюЖИђвАю ВџћВЋйьЋ┤ВБ╝ВёИВџћ. ВЮ┤ ьЁЇВіцьіИвіћ вЁ╝вгИВЮў {part}ВъЁвІѕвІц:\n\n{text}\n\nв░ўвЊюВІю ьЋюЖИђвАю ВџћВЋйьЋ┤ВБ╝ВІюЖ│а, ВўЂВќ┤ ВаёвгИВџЕВќ┤Ж░ђ ВъѕвІцвЕ┤ ьЋюЖИђ ВёцвфЁВЮё вЇДвХЎВЌгВБ╝ВёИВџћ.")
])

# ьћёвАгьћёьіИ, LLM, ВХюваЦ ьїїВёювЦ╝ ВЌ░Ж▓░ьЋўвіћ В▓┤ВЮИ ВЃЮВё▒
chain = prompt | llm | StrOutputParser()

# Word вгИВёю Ж░ЮВ▓┤ ВЃЮВё▒
doc = Document()
doc.add_heading('вЁ╝вгИ ВџћВЋй', 0)

# ьўёВъг ВІюЖ░ёВЮё ьїїВЮ╝вфЁВЌљ ьЈгьЋеВІюьѓцЖИ░ ВюёьЋю ьўЋВІЮ ВДђВаЋ

# Ж░Ђ ьїїьіИв│ёвАю ВџћВЋй ВІцьќЅ в░Ј Word вгИВёюВЌљ ВХћЖ░ђ
for i, text in enumerate(texts):
    print(f"ьїїьіИ {i+1}/2 ВџћВЋй ВцЉ...")
    # В▓┤ВЮИВЮё ВІцьќЅьЋўВЌг ВџћВЋй ВЃЮВё▒
    summary = chain.invoke({"text": text["content"], "part": text["part"]})
    
    # Word вгИВёюВЌљ ВџћВЋй вѓ┤ВџЕ ВХћЖ░ђ
    doc.add_heading(f'{text["part"]} ВџћВЋй', level=1)
    doc.add_paragraph(summary)
    doc.add_paragraph('-' * 40)  # ЖхгвХёВёа ВХћЖ░ђ
    
    # ВйўВєћВЌљ ВџћВЋй ВХюваЦ
    print(f"\n{text['part']} ВџћВЋй:")
    print(summary)
    print("-" * 80)

# ВЃЮВё▒вљю Word вгИВёювЦ╝ ВађВъЦ
output_path = f"вЁ╝вгИ_ВџћВЋй_{current_time}.docx"
doc.save(output_path)
print(f"\nьЋюЖИђ ВџћВЋйв│ИВЮ┤ {output_path}ВЌљ ВађВъЦвљўВЌѕВіхвІѕвІц")
