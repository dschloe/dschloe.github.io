    echo "Kibana is running!"
else
    echo "Kibana is NOT running!"
fi

Python мҪ”л“ң

  • нҢҢмқјлӘ…
# н•„мҡ”н•ң лқјмқҙлёҢлҹ¬лҰ¬л“Өмқ„ к°Җм ёмҳөлӢҲлӢӨ
import time                    # мӢңк°„ м§Җм—°мқ„ мң„н•ң лқјмқҙлёҢлҹ¬лҰ¬
import requests               # HTTP мҡ”мІӯмқ„ мң„н•ң лқјмқҙлёҢлҹ¬лҰ¬
from bs4 import BeautifulSoup # HTML нҢҢмӢұмқ„ мң„н•ң лқјмқҙлёҢлҹ¬лҰ¬
from elasticsearch import Elasticsearch  # Elasticsearch нҒҙлқјмқҙм–ёнҠё

# Elasticsearchм—җ кҙҖлҰ¬мһҗ кі„м •(elastic)мңјлЎң м—°кІ°н•©лӢҲлӢӨ
# localhost:9200мқҖ Elasticsearchмқҳ кё°ліё мЈјмҶҢмһ…лӢҲлӢӨ
es_admin = Elasticsearch("http://localhost:9200", basic_auth=("elastic", "123456"))
INDEX_NAME = "quotes"  # м ҖмһҘн•  мқёлҚұмҠӨ мқҙлҰ„мқ„ мғҒмҲҳлЎң м •мқҳ

# вң… Step 1: мқёлҚұмҠӨ мғқм„ұ (мқҙлҜё мЎҙмһ¬н•ҳл©ҙ мғқлһө)
# мқёлҚұмҠӨк°Җ мЎҙмһ¬н•ҳлҠ”м§Җ нҷ•мқён•ҳкі , м—Ҷмңјл©ҙ мғҲлЎң мғқм„ұн•©лӢҲлӢӨ
if not es_admin.indices.exists(index=INDEX_NAME):
    es_admin.indices.create(index=INDEX_NAME)  # мғҲ мқёлҚұмҠӨ мғқм„ұ
    print(f"вң… Index '{INDEX_NAME}' created.")
else:
    print(f"вң… Index '{INDEX_NAME}' already exists.")

# вң… Step 2: м—ӯн• (Role) мғқм„ұ (quotes_writer)
# quotes мқёлҚұмҠӨм—җ лҢҖн•ң мқҪкё°/м“°кё° к¶Ңн•ңмқ„ к°Җм§„ м—ӯн• мқ„ мғқм„ұн•©лӢҲлӢӨ
try:
    es_admin.security.put_role(
        name="quotes_writer",  # м—ӯн•  мқҙлҰ„
        body={
            "indices": [
                {
                    "names": [INDEX_NAME],  # м Ғмҡ©н•  мқёлҚұмҠӨ
                    "privileges": ["read", "write", "create_index", "auto_configure"]  # к¶Ңн•ң лӘ©лЎқ
                }
            ]
        }
    )
    print("вң… Role 'quotes_writer' created.")
except Exception as e:
    # м—ӯн• мқҙ мқҙлҜё мЎҙмһ¬н•ҳкұ°лӮҳ мғқм„ұм—җ мӢӨнҢЁн•ң кІҪмҡ° мҳҲмҷё мІҳлҰ¬
    print(f"вҡ пёҸ Role creation skipped or failed: {e}")

# вң… Step 3: мӮ¬мҡ©мһҗ(User) мғқм„ұ (sara)
# saraлқјлҠ” мӮ¬мҡ©мһҗлҘј мғқм„ұн•ҳкі  м—¬лҹ¬ м—ӯн• мқ„ л¶Җм—¬н•©лӢҲлӢӨ
try:
    es_admin.security.put_user(
        username="sara",  # мӮ¬мҡ©мһҗлӘ…
        body={
            "password": "230302",  # л№„л°ҖлІҲнҳё
            "roles": ["kibana_admin", "kibana_system", "quotes_writer"],  # л¶Җм—¬н•  м—ӯн• л“Ө
            "full_name": "Sara"  # м „мІҙ мқҙлҰ„
        }
    )
    print("вң… User 'sara' created.")
except Exception as e:
    # мӮ¬мҡ©мһҗк°Җ мқҙлҜё мЎҙмһ¬н•ҳкұ°лӮҳ мғқм„ұм—җ мӢӨнҢЁн•ң кІҪмҡ° мҳҲмҷё мІҳлҰ¬
    print(f"вҡ пёҸ User creation skipped or failed: {e}")

# вң… Step 4: sara кі„м •мңјлЎң мһ¬м—°кІ°
# мғҲлЎң мғқм„ұн•ң sara кі„м •мңјлЎң Elasticsearchм—җ лӢӨмӢң м—°кІ°н•©лӢҲлӢӨ
es = Elasticsearch("http://localhost:9200", basic_auth=("sara", "230302"))

# вң… Step 5: Quotes мҲҳм§‘ н•ЁмҲҳ м •мқҳ
def get_quotes():
    """
    quotes.toscrape.com мӣ№мӮ¬мқҙнҠём—җм„ң лӘ…м–ёл“Өмқ„ мҲҳм§‘н•ҳлҠ” н•ЁмҲҳ
    
    Returns:
        list: мҲҳм§‘лҗң лӘ…м–ё мҡ”мҶҢл“Өмқҳ лҰ¬мҠӨнҠё
    """
    res = requests.get("http://quotes.toscrape.com")  # мӣ№мӮ¬мқҙнҠём—җ GET мҡ”мІӯ
    soup = BeautifulSoup(res.text, "html.parser")     # HTMLмқ„ нҢҢмӢұ
    return soup.select(".quote")                       # .quote нҒҙлһҳмҠӨлҘј к°Җм§„ мҡ”мҶҢл“Өмқ„ м„ нғқ

# вң… Step 6: 30мҙҲл§ҲлӢӨ н•ҳлӮҳм”© м ҖмһҘ
# мҲҳм§‘лҗң лӘ…м–ёл“Өмқ„ к°Җм ёмҳөлӢҲлӢӨ
quotes = get_quotes()

# к°Ғ лӘ…м–ёмқ„ мҲңнҡҢн•ҳл©ҙм„ң Elasticsearchм—җ м ҖмһҘн•©лӢҲлӢӨ
for i, q in enumerate(quotes):
    # лӘ…м–ё лҚ°мқҙн„°лҘј л”•м…”л„ҲлҰ¬ нҳ•нғңлЎң кө¬м„ұн•©лӢҲлӢӨ
    doc = {
        "text": q.select_one(".text").text.strip(),      # лӘ…м–ё н…ҚмҠӨнҠё м¶”м¶ң
        "author": q.select_one(".author").text.strip(),  # м Җмһҗ мқҙлҰ„ м¶”м¶ң
        "tags": [tag.text for tag in q.select(".tag")]   # нғңк·ёл“Өмқ„ лҰ¬мҠӨнҠёлЎң м¶”м¶ң
    }
    
    # Elasticsearchм—җ л¬ём„ңлҘј м ҖмһҘн•©лӢҲлӢӨ
    res = es.index(index=INDEX_NAME, document=doc)
    print(f"[{i+1}] вң… Saved to Elasticsearch: {res['_id']}")
    
    # 30мҙҲ лҢҖкё° (лӢӨмқҢ лӘ…м–ё м ҖмһҘ м „)
    time.sleep(30)
